CREATE TABLE student(
    id varchar(11) primary key not null,
    name varchar(25) not null,
    major varchar(25) not null
);

CREATE TABLE prof_account(
    id varchar(11) primary key not null,
    name varchar(25) not null,
    major varchar(25) not null
);
CREATE TABLE stu_account(
    id varchar(11),
    foreign key(id) references student(id)
    on update cascade
    on delete cascade,
    password varchar(25) not null,
    name varchar(25),
    foreign key(name) references student(name)
    on update cascade
    on delete cascade,
    major varchar(25),
    foreign key(major) references student(major)
    on update cascade
    on delete cascade
);

CREATE TABLE prof_account(
    id varchar(11),
    foreign key(id) references professor(id)
    on update cascade
    on delete cascade,
    password varchar(25) not null,
    name varchar(25),
    foreign key(name) references professor(name)
    on update cascade
    on delete cascade,
    major varchar(25),
    foreign key(major) references professor(major)
    on update cascade
    on delete cascade
);

CREATE TABLE subject(
    id varchar(10) primary key not null,
    name varchar(25) not null,
    x double,
    y double,
    randomnum varchar(3),
    pid varchar(11),
    foreign key(pid) references prof_account(id)
    on update cascade
);

CREATE TABLE sugang(
    scode varchar(10),
    foreign key(scode) references subject(id)
    on update cascade,
    sid varchar(11),
    foreign key(sid) references stu_account(id)
    on update cascade
);
